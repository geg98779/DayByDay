## hand-slice js
- flat
```js
function myFlat(arr, count = 1) {
    let newArr = []
    for(let item of arr){
        if(Array.isArray(item) && count > 0){
            newArr.push(...myFlat(item, count - 1))
        }else{
            newArr.push(item)
        }
    }
    return newArr
}
```

- 去除个数最少的字母
```js
const deleteString = (s) => {
    let map = new Map()
    sArr = s.split('')
    let minCount = Infinity
    for(let i = 0; i < sArr.length; i++){
        if(map.has(sArr[i])){
            map.set(sArr[i],map.get(sArr[i]) + 1)
        }else{
            map.set(sArr[i], 1)
        }
        let count = map.get(sArr[i])
        if(minCount >= count){
            minCount = count
        }
    }
    let res = sArr.filter((item) => {
        return minCount !== map.get(item)
    })
    return res.join('')
}
```

- n个数组的组合
```js
const nNum = (args) => {
    const result = []
    const generateNum = (startIndex, eachResult) => {
        console.log(startIndex)
        if(eachResult.length === args.length){
            result.push(eachResult.slice())
            return
        }

        const startNum = args[startIndex]

        for(let i = 0; i < startNum.length; i++){
            eachResult.push(startNum[i])
            generateNum(startIndex + 1, eachResult)
            eachResult.pop()
        }
    }
    generateNum(0, [])
    return result
}
```
