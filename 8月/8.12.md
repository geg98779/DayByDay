## 力扣高频面试题 215. 数组中的第K个最大元素 快排 堆排
### 快排题解:
```js
//quick sort
function quickSelect(nums, k) {
    // Select a random pivot
    const pivotIndex = Math.floor(Math.random() * nums.length);
    const pivot = nums[pivotIndex];

    // Partition elements into three groups
    const big = [];
    const equal = [];
    const small = [];

    for (const num of nums) {
        if (num > pivot) {
            big.push(num);
        } else if (num < pivot) {
            small.push(num);
        } else {
            equal.push(num);
        }
    }

    // Recursively search based on the size of the partitions
    if (k <= big.length) {
        return quickSelect(big, k);
    }

    if (k > big.length + equal.length) {
        return quickSelect(small, k - big.length - equal.length);
    }

    // The kth largest element is in the 'equal' group, so it's the pivot
    return pivot;
}
var findKthLargest = function (nums, k) {
    return quickSelect(nums, k);
};
```

### 堆排序题解：
```js
//heap sort
function maxHeapify(nums, i, heapSize) {
    const left = 2 * i + 1;
    const right = left + 1;
    let largest = i;

    if (left < heapSize && nums[largest] < nums[left]) {
        largest = left;
    }
    if (right < heapSize && nums[largest] < nums[right]) {
        largest = right;
    }

    if (largest !== i) {
        swap(nums, i, largest);
        maxHeapify(nums, largest, heapSize);
    }
}

function swap(arr, i, j) {
    const temp = arr[i];
    arr[i] = arr[j];
    arr[j] = temp;
}

function initMaxHeap(nums) {
    for (let i = Math.floor(nums.length / 2) - 1; i >= 0; i--) {
        maxHeapify(nums, i, nums.length);
    }
}

var findKthLargest = function (nums, k) {
    initMaxHeap(nums);
    for (let i = 0; i < k - 1; i++) {
        swap(nums, 0, nums.length - 1 - i);
        maxHeapify(nums, 0, nums.length - 1 - i);
    }
    return nums[0];
};
```
