## 快捷键 F12 === ctrl + shift + j

```js
var threeSum = function(nums) {
    nums.sort((a,b) => a - b)
    const n = nums.length
    let ans = []
    for(let i = 0; i < n - 2; i++){
        const x = nums[i]
        if(i && nums[i] === nums[i-1]) continue
        // if(x + nums[i+1] + nums[i + 2] > 0) break
        // if(x + nums[n-1] + nums[n-2] < 0) continue
        let left = i + 1
        let right = n - 1
        while(left < right){
            const s = x + nums[left] + nums[right]
            if(s > 0){
                right--
            }else if(s < 0){
                left++
            }else{
                ans.push([x,nums[left],nums[right]])
                left++
                while(left < right && nums[left] === nums[left - 1]) left++
            }
        }
    }
    return ans
};
```
