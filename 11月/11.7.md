## foreach中 普通函数中的this为全局
```js
const obj = {
        name: 'MyObject',
        items: ['item1', 'item2', 'item3'],
        printItems: function () {
            console.log('Value of this inside printItems:', this); // 验证 this 的指向

            console.log('Using regular function without bind:');
            this.items.forEach(function (item) {
                console.log('Value of this inside forEach callback (regular function):', this);
                console.log(this.name + ': ' + item); // 这里会出错，因为 this 指向 window/undefined
            });
        }
    };

    obj.printItems();
```
