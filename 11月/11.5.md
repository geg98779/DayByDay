## 监听系统主题变更解决方案 window.matchMedia('(prefers-color-scheme: dark)')
```js
// 监听系统主题变更
let matchMedia
const watchSystemTheme = () => {
  // 仅需一次初始化
  if (matchMedia) return
  matchMedia = window.matchMedia('(prefers-color-scheme: dark)')
  // 监听主题变更
  matchMedia.onchange = () => {
    changeTheme(THEME_SYSTEM)
  }
}
```
