## hand-slice js
- 使用node起一个httpServer
```js
const http = require('http')
const server = http.createServer((req,res) => {
    res.writeHead(200, {})
    res.end('hello world')
})

const port = 3000
server.listen(port, () => {
    console.log('start')
})
```

- 爬楼梯
```js
const step = (n) => {
    const dp = []
    dp[1] = 1
    dp[2] = 2
    for(let i = 3; i <= n; i++){
        dp[i] = dp[i - 1] + dp[i - 2]
    }
    return dp[n]
}
```
