## promise.retry
```js
function promiseRetry(promiseFn, maxAttempts, delay){
    return new Promise((resolve, reject) => {
        const retry = (attempt) => {
            promiseFn().then(resolve).catch(error => {
                if(attempt < maxAttempts){
                    console.log(`Attempt ${attempt} failed. Retrying in ${delay}ms...`)
                    setTimeout(() => {
                        retry(attempt + 1)
                    }, delay)
                }else{
                    reject(`Max attempts reached. Last error: ${error}`)
                }
            })
        }
        retry(1)
    })
}
```

## promisify
```js
function promisify(fn){
    return function(...args){
        return new Promise((resolve, reject) => {
            fn(...args, (err,result) => {
                if(err) {
                    reject(err)
                }else{
                    resolve(result)
                }
            })
        })
    }
}
```
