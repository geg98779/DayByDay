## hand-slice js
- 括号匹配
```js
const match = (n) => {
    let res = []
    const backShaking = (str, left, right) => {
        if(str.length === n * 2){
            res.push(str.slice())
            return
        }
        for(let i = 0; i < n; i++){
            if(left > 0){
                backShaking(str + '(', left - 1, right)
            }
            if(right > left){
                backShaking(str + ')', left, right - 1)
            }
        }
    }
    backShaking('',n,n)
    return res
}
```
