## promise.all
```js
const myall = (data) => {
            return new Promise((resolve, reject) => {
                let result = []
                let count = 0
                const addData = (item, index) => {
                    result[index] = item
                    count++
                    if(result.length === data.length){
                        resolve(result)
                    }
                }
                data.forEach((item,index) => {
                    if(item instanceof promise){
                        item.then((res) => addData(res, index), (err) => reject(err))
                    }else{
                        addData(item, index)
                    }
                })
            })
        }
```
